async function generateIndividualId(conceptId, trigUri) {

    // –í—ã–ø–æ–ª–Ω—è–µ–º –∑–∞–ø—Ä–æ—Å

    // –ï—Å–ª–∏ –∏–Ω–¥–∏–≤–∏–¥–æ–≤ –Ω–µ—Ç ‚Äî –ø–µ—Ä–≤—ã–π –∏–Ω–¥–∏–≤–∏–¥ –±–µ–∑ —Å—É—Ñ—Ñ–∏–∫—Å–∞
    if (existingIndividuals.length === 0) {

    }

    // –ü—Ä–æ–≤–µ—Ä—è–µ–º, –µ—Å—Ç—å –ª–∏ –∏–Ω–¥–∏–≤–∏–¥ –±–µ–∑ —Å—É—Ñ—Ñ–∏–∫—Å–∞ (conceptId –Ω–∞–ø—Ä—è–º—É—é)

        return localName === conceptId;
    });

    // –ï—Å–ª–∏ –±–∞–∑–æ–≤—ã–π ID –µ—â—ë –Ω–µ –∑–∞–Ω—è—Ç ‚Äî –∏—Å–ø–æ–ª—å–∑—É–µ–º –µ–≥–æ
    if (!hasBasicId) {
        return conceptId;
    }

    // –ù–∞—Ö–æ–¥–∏–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Å—É—â–µ—Å—Ç–≤—É—é—â–∏–π –Ω–æ–º–µ—Ä
    let maxNumber = 1;
    const pattern = new RegExp(`^${escapeRegExp(conceptId)}${SEPARATOR}(\\d+)$`);

    existingIndividuals.forEach(ind => {
        const localName = getLocalName(ind.uri);
        const match = localName.match(pattern);
        if (match) {

    // –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å–ª–µ–¥—É—é—â–∏–π –Ω–æ–º–µ—Ä

/**

function getLocalName(uri) {
    const hashIndex = uri.lastIndexOf('#');
    if (hashIndex !== -1) {
        return uri.substring(hashIndex + 1);
    }
    const slashIndex = uri.lastIndexOf('/');
    if (slashIndex !== -1) {
        return uri.substring(slashIndex + 1);
    }
    return uri;
}

}
